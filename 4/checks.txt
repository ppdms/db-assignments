SELECT id
FROM genre
WHERE id not in (
    SELECT genre_id
    FROM hasGenre
)

SELECT id
FROM productioncompany
WHERE id not in (
    SELECT pc_id
    FROM hasProductioncompany
)

SELECT id
FROM Keyword
WHERE id not in (
    SELECT keyword_id
    FROM haskeyword
)

FROM movie
WHERE id in (
    SELECT movie_id
    FROM ratings
)
